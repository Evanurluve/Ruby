<script>

const $ = id => document.getElementById(id);


const textGrid   = $('textGrid');

const displayText= $('displayText');

const textColor  = $('textColorPicker');

const wghtMin    = $('wghtMin');

const wghtMax    = $('wghtMax');

const phaseShift = $('phaseShift');

const rowOffset  = $('rowOffset');

const charOffset = $('charOffset');

const fontSize   = $('fontSize');

const colGap     = $('colGap');

const lineHeight = $('lineHeight');

const rowCountEl = $('rowCount');

const animateToggle = $('animateToggle');

const captureArea = $('captureArea');


let animPhase = 0;

let animId = null;

let cells = []; // {span, r, c}

let currentRows = 0, currentCols = 0;


const DEFAULTS = {

  displayText:'BREATHE OUT', textColor:'#ffffff', wghtMin:'100', wghtMax:'900',

  phaseShift:4.5, rowOffset:0.7, charOffset:0.2, fontSize:75, colGap:0.74,

  lineHeight:0.88, rowCount:15

};


// 只在文本/行数变化时重建 DOM 结构，其余参数变化只更新已有节点的样式（避免卡顿）

function buildGrid(){

  const text = displayText.value || ' ';

  const rows = parseInt(rowCountEl.value);

  textGrid.innerHTML = '';

  cells = [];

  currentRows = rows;

  currentCols = text.length;


  for(let r=0; r<rows; r++){

    const rowEl = document.createElement('div');

    rowEl.className = 'row';

    const group = document.createElement('div');

    group.className = 'word-group';

    for(let c=0; c<text.length; c++){

      const span = document.createElement('span');

      span.className = 'char';

      span.innerText = text[c];

      group.appendChild(span);

      cells.push({span, r, c});

    }

    rowEl.appendChild(group);

    textGrid.appendChild(rowEl);

  }

}


// 每帧/每次参数变化只更新样式，不重建 DOM —— 这是性能关键

function updateStyles(){

  const phase   = parseFloat(phaseShift.value) + animPhase;

  const rOff    = parseFloat(rowOffset.value);

  const cOff    = parseFloat(charOffset.value);

  const min     = parseInt(wghtMin.value);

  const max     = parseInt(wghtMax.value);

  const range   = max - min;


  document.documentElement.style.setProperty('--text', textColor.value);

  textGrid.style.setProperty('--font-size', fontSize.value + 'px');

  textGrid.style.setProperty('--line-height', lineHeight.value);

  textGrid.style.setProperty('--col-gap', colGap.value);


  for(const cell of cells){

    const angle = phase + (cell.r * rOff) + (cell.c * cOff);

    const sine = (Math.sin(angle) + 1) / 2;

    const weight = Math.round(min + sine * range);

    cell.span.style.fontVariationSettings = `'wght' ${weight}`;

    cell.span.style.color = textColor.value;

    cell.span.style.opacity = 0.15 + sine * 0.85;

  }

}


function updateLabels(){

  $('val-phase').innerText  = parseFloat(phaseShift.value).toFixed(2);

  $('val-row').innerText    = parseFloat(rowOffset.value).toFixed(2);

  $('val-char').innerText   = parseFloat(charOffset.value).toFixed(2);

  $('val-size').innerText   = fontSize.value + 'px';

  $('val-colgap').innerText = parseFloat(colGap.value).toFixed(2);

  $('val-lh').innerText     = parseFloat(lineHeight.value).toFixed(2);

  $('val-rows').innerText   = rowCountEl.value;

}


function fullRender(){

  buildGrid();

  updateLabels();

  updateStyles();

}


function lightUpdate(){

  updateLabels();

  updateStyles();

}


// 结构性变化 -> 重建 DOM

[displayText, rowCountEl].forEach(el => el.addEventListener('input', fullRender));


// 纯样式变化 -> 只更新已有节点，不重建（流畅的关键）

[textColor, wghtMin, wghtMax, phaseShift, rowOffset, charOffset, fontSize, colGap, lineHeight]

  .forEach(el => el.addEventListener('input', lightUpdate));


// 极缓自动位移动画

function animate(){

  animPhase += 0.006;

  updateStyles();

  animId = requestAnimationFrame(animate);

}

animateToggle.addEventListener('change', () => {

  if(animateToggle.checked){

    animId = requestAnimationFrame(animate);

  }else{

    cancelAnimationFrame(animId);

    animPhase = 0;

    updateStyles();

  }

});


// 还原默认

$('resetBtn').addEventListener('click', () => {

  displayText.value = DEFAULTS.displayText;

  textColor.value = DEFAULTS.textColor;

  wghtMin.value = DEFAULTS.wghtMin;

  wghtMax.value = DEFAULTS.wghtMax;

  phaseShift.value = DEFAULTS.phaseShift;

  rowOffset.value = DEFAULTS.rowOffset;

  charOffset.value = DEFAULTS.charOffset;

  fontSize.value = DEFAULTS.fontSize;

  colGap.value = DEFAULTS.colGap;

  lineHeight.value = DEFAULTS.lineHeight;

  rowCountEl.value = DEFAULTS.rowCount;

  fullRender();

});


// 透明背景导出

$('exportPngBtn').addEventListener('click', () => {

  const prevBg = captureArea.style.backgroundColor;

  captureArea.style.backgroundColor = 'transparent';

  html2canvas(captureArea, { backgroundColor:null, scale:2, logging:false })

    .then(canvas => {

      const link = document.createElement('a');

      link.download = 'typography_transparent.png';

      link.href = canvas.toDataURL('image/png');

      link.click();

      captureArea.style.backgroundColor = prevBg || 'var(--bg)';

    });

});


fullRender();

</script>
